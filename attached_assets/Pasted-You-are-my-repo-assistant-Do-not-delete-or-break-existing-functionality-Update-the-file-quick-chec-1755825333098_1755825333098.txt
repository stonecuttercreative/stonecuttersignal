#!/usr/bin/env python3
"""
Quick validation script for Stonecutter Signal extensions.
Supports both run_signal_engine and run_signal, flat or nested outputs.
"""

import json
import sys

def _import_runner():
    try:
        from stonecutter import run_signal_engine as runner
        return runner
    except Exception:
        pass
    try:
        from stonecutter import run_signal as runner
        return runner
    except Exception:
        pass
    try:
        from stonecutter.engine import run_signal as runner
        return runner
    except Exception:
        print("❌ Could not import run_signal_engine or run_signal.", file=sys.stderr)
        raise

def _scores_block(result):
    if isinstance(result, dict) and "signal_scores" in result:
        ss = result["signal_scores"] or {}
        return ss.get("scores", {}), ss.get("notes", {})
    return result.get("scores", {}), result.get("notes", {})

def _has_audience(result, audience_text):
    if not isinstance(result, dict):
        return False
    if "audience_note" in result and result["audience_note"]:
        return audience_text.lower() in str(result["audience_note"]).lower()
    return audience_text.lower() in json.dumps(result).lower()

def test_legacy_string_input(run):
    print("Test 1: Legacy plain string input...")
    brief = "Concept about clarity in crypto ads"
    result = run(brief)
    scores, notes = _scores_block(result)
    assert "conversation_fit" in scores, f"conversation_fit missing. Got keys: {list(scores.keys())}"
    assert "distribution_fit" not in scores, "distribution_fit should be absent"
    assert "distribution_fit" in (notes or {}), f"notes missing distribution_fit. Notes: {notes}"
    assert "not evaluated" in str(notes["distribution_fit"]).lower(), f"Wrong note: {notes['distribution_fit']}"
    print("✓ Test 1 passed")

def test_dict_input_audience_only(run):
    print("Test 2: Dict input with audience only...")
    audience = "privacy-conscious millennials"
    brief = {
        "brand": "Acme",
        "category": "Finance",
        "concept": "Calm campaign during market noise",
        "mode": "realtime",
        "audience": audience
    }
    result = run(brief)
    scores, notes = _scores_block(result)
    assert "conversation_fit" in scores, "conversation_fit missing"
    assert "distribution_fit" not in scores, "distribution_fit should be absent"
    assert _has_audience(result, audience), "Audience text not found in output"
    assert "distribution_fit" in (notes or {}), "Expected distribution_fit note"
    print("✓ Test 2 passed")

def test_dict_input_audience_and_channels(run):
    print("Test 3: Dict input with audience and channels...")
    brief = {
        "brand": "Tesla",
        "category": "Automotive",
        "concept": "Sustainable transport message",
        "mode": "realtime",
        "audience": "eco-conscious professionals",
        "channels": "TV, YouTube, Instagram"
    }
    result = run(brief)
    scores, _ = _scores_block(result)
    assert "conversation_fit" in scores, "conversation_fit missing"
    assert "distribution_fit" in scores, "distribution_fit missing"
    dist = scores["distribution_fit"]
    assert isinstance(dist, (int, float)) and 0 <= dist <= 100, f"distribution_fit out of range: {dist}"
    print("✓ Test 3 passed")

def main():
    print("Running Stonecutter Signal validation checks...\n")
    run = _import_runner()
    try:
        test_legacy_string_input(run)
        test_dict_input_audience_only(run)
        test_dict_input_audience_and_channels(run)
        print("\n🎉 All checks passed")
    except AssertionError as e:
        print(f"\n❌ Check failed:\n{e}")
        print("\nTip: print the full output of the failing case to inspect shape differences.")

if __name__ == "__main__":
    main()
