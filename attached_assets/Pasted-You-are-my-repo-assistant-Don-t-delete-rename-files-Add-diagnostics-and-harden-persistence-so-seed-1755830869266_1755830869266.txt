  # BEGIN stonecutter extension: sqlite harden
  import sqlite3, json, time, uuid, os
  from ..settings import settings
  from ..logging_conf import logger

  _CONN = None
  def init_db():
      global _CONN
      if _CONN: return _CONN
      os.makedirs(os.path.dirname(settings.db_path), exist_ok=True)
      _CONN = sqlite3.connect(settings.db_path, isolation_level=None, check_same_thread=False)
      cur = _CONN.cursor()
      cur.execute("""CREATE TABLE IF NOT EXISTS runs(
        id TEXT PRIMARY KEY, ts INTEGER, mode TEXT, brand TEXT, category TEXT,
        audience TEXT, channels TEXT, confidence INTEGER, consensus INTEGER, diversity INTEGER, story TEXT
      )""")
      cur.execute("""CREATE TABLE IF NOT EXISTS scores(
        run_id TEXT, key TEXT, value INTEGER
      )""")
      cur.execute("""CREATE TABLE IF NOT EXISTS providers(
        run_id TEXT, name TEXT, model TEXT, latency_ms INTEGER, is_mock INTEGER
      )""")
      return _CONN

  def save_run(rec: dict):
      try:
          conn = init_db(); cur = conn.cursor()
          cur.execute("INSERT OR REPLACE INTO runs VALUES (?,?,?,?,?,?,?,?,?,?,?)", (
              rec["id"], rec["ts"], rec.get("mode"), rec.get("brand"), rec.get("category"),
              rec.get("audience"), rec.get("channels"), int(rec["meta"]["signal_scores"]["confidence"]),
              int(rec["meta"]["signal_scores"]["consensus"]), int(rec["meta"]["signal_scores"]["diversity"]),
              (rec.get("story") or "")[:280]
          ))
          for k,v in rec.get("scores", {}).items():
              cur.execute("INSERT INTO scores(run_id,key,value) VALUES(?,?,?)", (rec["id"], k, int(v)))
          for p in rec.get("providers", []):
              cur.execute("INSERT INTO providers(run_id,name,model,latency_ms,is_mock) VALUES(?,?,?,?,?)", (
                  rec["id"], p.get("provider"), p.get("model"), p.get("latency_ms") or 0,
                  1 if (p.get("model") is None) else 0
              ))
      except Exception as e:
          logger.error(f"[persistence] save_run failed: {e}")
  def last_runs(limit=50):
      conn = init_db(); cur = conn.cursor()
      cur.execute("SELECT id, ts, brand, category, audience, channels, confidence, consensus, diversity FROM runs ORDER BY ts DESC LIMIT ?", (limit,))
      rows = cur.fetchall()
      return [
        {"id": r[0], "ts": r[1], "brand": r[2], "category": r[3], "audience": r[4], "channels": r[5],
         "confidence": r[6], "consensus": r[7], "diversity": r[8]}
        for r in rows
      ]
  # END stonecutter extension: sqlite harden
