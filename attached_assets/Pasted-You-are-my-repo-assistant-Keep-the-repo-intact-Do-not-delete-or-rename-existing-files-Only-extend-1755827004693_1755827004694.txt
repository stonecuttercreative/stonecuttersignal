# BEGIN stonecutter extension: real Claude provider
from typing import Dict, Any
import json, asyncio, time
from anthropic import Anthropic
from ..settings import settings
from ._mock_util import mock_complete

PROMPT_WRAPPER = """You are a cultural diagnostics analyst.
Return STRICT JSON ONLY, no prose.
JSON schema:
{{
  "cluster": str,
  "archetype": str,
  "story": str,
  "scores": {{
    "cultural_fit": int, "clarity": int, "emotional_resonance": int,
    "differentiation": int, "conversation_fit": int{extra_score}
  }},
  "claims": [str, str, str]
}}
Input:
{input_block}
"""

class ClaudeProvider:
    name = "claude"

    async def complete(self, prompt: str, **kw) -> Dict[str, Any]:
        # Soft fallback if disabled or no key
        if not settings.enable_claude or not settings.anthropic_api_key:
            return await mock_complete(self.name, prompt)

        # Build wrapped prompt; include distribution_fit only when channels provided
        extra = ', "distribution_fit": int' if ("Channels:" in prompt and "N/A" not in prompt) else ""
        wrapped = PROMPT_WRAPPER.format(extra_score=extra, input_block=prompt)

        # Call Claude using the Messages API
        start = time.time()
        try:
            client = Anthropic(api_key=settings.anthropic_api_key)
            # Claude is not async; run in thread to avoid blocking
            def _call():
                msg = client.messages.create(
                    model=settings.claude_model,
                    max_tokens=800,
                    temperature=0.6,
                    messages=[{"role": "user", "content": wrapped}],
                )
                return msg
            loop = asyncio.get_running_loop()
            msg = await loop.run_in_executor(None, _call)

            # Extract text content
            content = "".join([b.text for b in msg.content if getattr(b, "type", "") == "text"])
            # Parse JSON strictly; if parse fails, fallback to mock
            try:
                data = json.loads(content)
            except Exception:
                return await mock_complete(self.name, prompt)

            # Telemetry
            data["_telemetry"] = {
                "provider": self.name,
                "latency_ms": int((time.time() - start) * 1000),
                "model": settings.claude_model,
                # token usage not always present; include when available
                "input_tokens": getattr(getattr(msg, "usage", None), "input_tokens", None),
                "output_tokens": getattr(getattr(msg, "usage", None), "output_tokens", None),
            }
            return data
        except Exception:
            # Never hard-fail the run; fallback to mock
            return await mock_complete(self.name, prompt)
# END stonecutter extension

# BEGIN stonecutter extension: compare script
import asyncio, json
from stonecutter.engine import run_signal
from stonecutter.providers.registry import load_panel
from stonecutter.settings import settings
from stonecutter.engine import build_prompt  # if exposed; otherwise inline prompt build

async def _panel(prompt: str):
    panel = load_panel()
    outs = []
    for m in panel:
        r = await m.complete(prompt)
        r["_provider_name"] = getattr(m, "name", "unknown")
        outs.append(r)
    return outs

def main():
    brief = {
        "brand":"Acme","category":"Finance","concept":"Clarity-first crypto education",
        "mode":"realtime","audience":"Adults 25-45 seeking stability","channels":"YouTube, CTV"
    }
    result = asyncio.run(run_signal(brief))
    print("\n=== ARBITRATED RESULT ===")
    print(json.dumps(result, indent=2, ensure_ascii=False))

    # Show raw provider contributions (names + top-level scores)
    prompt = build_prompt(type("x",(object,),brief)()) if callable(getattr(build_prompt,"__call__",None)) else str(brief)
    raw = asyncio.run(_panel(prompt))
    print("\n=== PROVIDER CONTRIBUTIONS (first 2 shown if many) ===")
    for r in raw[:6]:
        name = r.get("_provider_name")
        scores = r.get("scores", {})
        print(f"- {name}: cultural_fit={scores.get('cultural_fit')} clarity={scores.get('clarity')} conversation_fit={scores.get('conversation_fit')}")

if __name__ == "__main__":
    main()
# END stonecutter extension
